Überblick pro Dokument

SELECT
    nd.title,
    nd.source_type,
    COUNT(nc.id)                                          AS chunks,
    SUM(CASE WHEN nc.embedding IS NOT NULL THEN 1 END)    AS mit_embedding,
    SUM(CASE WHEN nc.abbrev_map IS NOT NULL THEN 1 END)   AS mit_abbrev,
    nc.doc_class                                          AS klasse,
    nd.ingest_ts::date                                    AS datum
FROM norm_documents nd
LEFT JOIN norm_chunks nc ON nc.doc_id = nd.id
GROUP BY nd.title, nd.source_type, nc.doc_class, nd.ingest_ts::date
ORDER BY nd.ingest_ts DESC;


Chunk-Typen je Dokument:
SELECT
    nd.title,
    nc.chunk_type,
    COUNT(*) AS anzahl
FROM norm_chunks nc
JOIN norm_documents nd ON nc.doc_id = nd.id
GROUP BY nd.title, nc.chunk_type
ORDER BY nd.title, anzahl DESC;


Welche Abkürzungen wurden aufgelöst?
SELECT DISTINCT
    elem->>'abbrev'   AS abkürzung,
    elem->>'resolved' AS auflösung,
    elem->>'label'    AS label,
    COUNT(*)          AS in_chunks
FROM norm_chunks,
     jsonb_array_elements(abbrev_map) AS elem
GROUP BY abkürzung, auflösung, label
ORDER BY in_chunks DESC;

Chunks mit Originaltext prüfen:
SELECT
    nc.norm_reference,
    nc.chunk_type,
    LEFT(nc.content, 120)          AS text_aufgelöst,
    LEFT(nc.content_original, 120) AS text_original
FROM norm_chunks nc
JOIN norm_documents nd ON nc.doc_id = nd.id
WHERE nd.title = 'Hundegesetz'
  AND nc.abbrev_map IS NOT NULL
LIMIT 10;

Fehlerhafte Jobs:
SELECT filename, status, error_message, started_at
FROM ingest_jobs
WHERE status = 'error'
ORDER BY started_at DESC;

Export als CSV
docker exec -i mnr-postgres psql -U mnr -d mnr_db << 'SQL'
\COPY (
    SELECT
        nd.title,
        nd.source_type,
        nc.doc_class,
        nc.chunk_type,
        nc.norm_reference,
        nc.confidence_weight,
        nc.token_count,
        LEFT(nc.content, 200) AS text_vorschau
    FROM norm_chunks nc
    JOIN norm_documents nd ON nc.doc_id = nd.id
    ORDER BY nd.title, nc.hierarchy_level
) TO '/tmp/ingest_auswertung.csv' CSV HEADER DELIMITER ';'
